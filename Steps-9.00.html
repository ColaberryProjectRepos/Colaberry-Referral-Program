Measures Calculations

-   see the below GIF for more understanding 

 1.     Total Cost = [Total payOut] + [Program Cost]

2.     Total Enrolled = CALCULATE(COUNT(Referee[ReferralName]),FILTER(Referee,Referee[RFM_Enrolled]=1))

3.     Total payOut = [First Reward]+[Second Reward]

4.     Total sign ups = CALCULATE(Count('Members'[MemberName]))

5.     Total Tuition = ([Cnt of tution paid])*1999

6.     Member Sign Up per Month = DIVIDE ([Total sign ups], DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

 

7.     Indicator sign Ups = If ([Referrals per Month QoQ%]<0,UNICHAR(9660),UNICHAR(9650))

 

8.     Referral Enrollment per Month = DIVIDE ([Total Enrolled], DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

 

9.     Indicator Enrollement Ups = If([Member Enrollment QoQ%]<0,UNICHAR(9660),UNICHAR(9650))

 

10. Referral Payout per Month = DIVIDE ([Total payOut], DISTINCTCOUNT('Date'[YearMonthShort]),0) +0

 

11. Indicator Payouts = If([Member Sign Up per Month]<0,UNICHAR(9680),UNICHAR(9600))

 

12. ROI per Month = Divide (([Total Tuition] - [Program Cost] - [Total payOut]), DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

13. Indicator ROI Ups = If([ROI QoQ%]<0,UNICHAR(9660),UNICHAR(9650))

 

14. Referrals per Month QoQ% =

15. VAR __PREV_QUARTER = CALCULATE ([Member Sign Up per Month], DATEADD('Date'[Date], -1, QUARTER))

RETURN

DIVIDE ([Member Sign Up per Month] - __PREV_QUARTER, __PREV_QUARTER)

 

16. Member Sign Up per Month = DIVIDE ([Total sign ups], DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

 

17. Member Enrollment QoQ% =

VAR __PREV_QUARTER = CALCULATE ([Referral Enrollment per Month], DATEADD('Date'[Date], -1, QUARTER))

RETURN

DIVIDE ([Referral Enrollment per Month] - __PREV_QUARTER, __PREV_QUARTER)

 

18. Referral Enrollment per Month = DIVIDE ([Total Enrolled], DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

 

19. Referral Payout per Month = DIVIDE ([Total payOut], DISTINCTCOUNT('Date'[YearMonthShort]),0) +0

 

20. Referral Payout per Month QoQ% =

VAR __PREV_Quarter = CALCULATE ([Referral Payout per Month], DATEADD('Date'[Date], -1, QUARTER))

RETURN

DIVIDE ([Referral Payout per Month] - __PREV_Quarter, __PREV_Quarter) +0

 

21. ROI per Month = Divide (([Total Tuition] - [Program Cost] - [Total payOut]), DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

 

22. ROI QoQ% =

VAR __PREV_QUARTER = CALCULATE ([ROI per Month], DATEADD('Date'[Date], -1, QUARTER))

RETURN

DIVIDE ([ROI per Month] - __PREV_QUARTER, __PREV_QUARTER)

 

23. ThisMth vs AVGMthlySignup =

DIVIDE ([Cnt Members], [AVG Member Per Month],0)

 

24. Member Sign Up per Month = DIVIDE ([Total sign ups], DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

 

25. AVG Member Per Month = DIVIDE (Count ('Referral Program Table'[MemberName]), DISTINCTCOUNT ('Referral Program Table'[RegDate MonthYear]),0) + 0

 

 

26. AVG Time for 1st Reward =

VAR NumberSignUp = CALCULATE (Count ('Referral Program Table'[MemberName]),

FILTER ('Referral Program Table','Referral Program Table'[.Reward 1]>0))

RETURN

ROUNDUP (DIVIDE (Sum ('Referral Program Table'[Elapse Time before 1st reward]),NumberSignUp),0) &" Days"

 

27. ThisMth vs AVGMthlyEnrollment =

DIVIDE ([Total Enrolled], [AVG Enrolled Per Month],0)

 

28. Referral Enrollment per Month = DIVIDE ([Total Enrolled], DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

 

29. AVG Enrolled Per Month = VAR Enrolled = CALCULATE (Count ('Referral Program Table'[ReferralName]),

FILTER ('Referral Program Table','Referral Program Table'[RFM_Enrolled] = 1))

Var NberofMonth = DISTINCTCOUNT ('Referral Program Table'[RegDate MonthYear])

Var AvgperMonth = DIVIDE (Enrolled,NberofMonth,0) + 0

Return

AvgperMonth

 

30. % Complete = AVERAGE (Referee [Number Completed])/6

 

31. ThisMth vs AVGMthlyPayout =

DIVIDE ([Total payOut], [Avg Payout Per Month],0)

 

32. Referral Payout per Month = DIVIDE ([Total payOut], DISTINCTCOUNT('Date'[YearMonthShort]),0) +0

 

 

33. Avg Payout Per Month = DIVIDE (([.Reward 1]+[.Reward 2]),DISTINCTCOUNT('Referral Program Table'[RegDate MonthYear]),0)

 

34. ThisMth vs AVGMthlyROI =

DIVIDE ([ROI per Month], [AVG ROI Per Month],0)

 

35. ROI per Month = Divide (([Total Tuition] - [Program Cost] - [Total payOut]), DISTINCTCOUNT('Date'[YearMonthShort]),0) + 0

 

36. AVG ROI Per Month = DIVIDE ([.ROI ],DISTINCTCOUNT('Referral Program Table'[RegDate MonthYear]),0) + 0

37. Program Cost = Count('Date'[Date]) *8.21

38. First Reward = CALCULATE(COUNT('Commission1'[MemberName]),FILTER(Commission1,Commission1[RFM_Qualified]=1))*50

39. Second Reward = CALCULATE (Count (Commission2[MemberName]),FILTER (Commission2,Commission2[RFM_Approved]=1))*150
